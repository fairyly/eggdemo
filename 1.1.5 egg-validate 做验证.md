# egg-validate


## install

```
npm i egg-validate --save
```

## 配置

```
// config/plugin.js
exports.validate = {
  enable: true,
  package: 'egg-validate',
};
```

## 使用

```
# controller 中校验

// 定义创建接口的请求参数规则
const rule = {
  accesstoken: 'string',
  title: 'string',
  tab: { type: 'enum', values: [ 'ask', 'share', 'job' ], required: false },
  content: 'string',
};


// 校验 `ctx.request.body` 是否符合我们预期的格式
// 如果参数校验未通过，将会抛出一个 status = 422 的异常
ctx.validate(createRule, ctx.request.body);
```
